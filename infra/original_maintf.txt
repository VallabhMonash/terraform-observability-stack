resource "docker_network" "observability" {
  name = "observability-net"
}

resource "docker_image" "app" {
  name = "obs-app:latest"

  build {
    context    = "${path.module}/../app"
    dockerfile = "Dockerfile"
  }
}

resource "docker_container" "app" {
  name  = "obs-app"
  image = docker_image.app.image_id

  networks_advanced {
    name = docker_network.observability.name
  }

  ports {
    internal = 8000
    external = 8000
  }
}

resource "docker_image" "prometheus" {
  name = "prom/prometheus:latest"
}


resource "local_file" "prometheus_config" {
  filename = abspath("${path.module}/prometheus.yml")
  content  = <<-YAML
global:
  scrape_interval: 5s

scrape_configs:
  - job_name: "obs-app"
    static_configs:
      - targets: ["${docker_container.app.name}:8000"]
YAML
}

resource "docker_container" "prometheus" {
  name  = "prometheus"
  image = docker_image.prometheus.image_id

  networks_advanced {
    name = docker_network.observability.name
  }

  ports {
    internal = 9090
    external = 9090
  }

  volumes {
    host_path      = abspath(local_file.prometheus_config.filename)
    container_path = "/etc/prometheus/prometheus.yml"
  }

  command = ["--config.file=/etc/prometheus/prometheus.yml"]
}

resource "docker_image" "grafana" {
  name = "grafana/grafana:latest"
}

resource "docker_container" "grafana" {
  name  = "grafana"
  image = docker_image.grafana.image_id

  networks_advanced {
    name = docker_network.observability.name
  }

  ports {
    internal = 3000
    external = 3000
  }

  env = [
    "GF_SECURITY_ADMIN_USER=admin",
    "GF_SECURITY_ADMIN_PASSWORD=admin"
  ]
}
